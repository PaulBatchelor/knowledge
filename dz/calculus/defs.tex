\begdef{Rate of Change and Slope of the Tangent Line}
\enddef

The {\bf average rate of change} in $f$ on the interval
$[a, x]$ is the slope of the corresponding secant line:

$$\
m_{sec} = {f(x) - f(a) \over x - a}
$$

The {\bf instantaneous rate of change} in $f$ at $a$ is

$$\
m_{tan} = \lim_{x \to a} {
f(x) - f(a)
\over
x - a
}
$$

Which is also the {\bf slope of the tangent line} at $(a, f(a))$,
provided this limit exists. The {\bf tangent line} is the unique
line through $(a, f(a))$ with sloep $m_{tan}$. Its equation is:

$$\
y - f(a) = m_{tan}(x- a)
$$

\begdef {Alternate Definition: Rate of Change and the Slope of the Tangent Line}
The {\bf average rate of change} of $f$ on the interval $[a, a + h]$
is the slope of the corresponding secant line:

$$\
m_{sec} = {
f(a + h) - f(a)
\over
h
}
$$

The {\bf instantaneous rate of change} in $f$ at $a$ is

$$\
m_{tan} = \lim_{h \to 0}{
f(a + h) - f(a)
\over
h
}
$$

which is also the {\bf slope of the tangent line} at $(a, f(a))$,
provided this limit exists.

\enddef

\begdef{The derviative of a function at a point}
The {\bf derivative of $f$ at $a$}, denoted $f'(a)$, is given
by either of the two following limits, provided the limits exist and $a$ is in the
domain of $f$:

$$\
f'(a) = \lim_{x \to a}{
f(x) - f(a)
\over
x - a
}
$$

or

$$\
f'(a) = \lim_{h \to 0}{
f(a + h) - f(a)
\over
h
}
$$

\enddef

\begdef{The Derivative Function}
The {\bf derivative} is the function

$$\
f'(x) = \lim_{h \to 0}{
f(x + h) - f(x)
\over
h
}
$$

provided the limit exists and $x$ is in the domain of $f$.
If $f'(x)$ exists, we say that $f$ is {\bf differentiable}
at $x$. If $f$ is differentiable at every point of an open
interval $I$, we say that $f$ is differentiable from on $I$.

\enddef

\begdef{The Number $e$}
The number $e = 2.718281828459...$ satisfies

$$\
\lim_{h \to 0} {
e^h - 1
\over h
} = 1
$$

It is the base of the natural exponential function $f(x) = e^x$.
\enddef

\begdef{Higher-Order Derivatives}
Assuming $y = f(x)$ can be differentiated as often as necessary,
the {\bf second derivative} of $f$ is

$$\
f''(x) = {d \over dx}(f'(x)).
$$

For integers $n \ge 1$, the {\bf {\it n}th derivative} of $f$ is

$$\
f^{(n)}(x) = {d \over dx}(f^{(n - 1)}(x)).
$$

\enddef

\begdef{Limit of a Function (preliminary)}
Suppose the function $f$ is defined for all $x$ near $a$
except possibly at $a$. if $f(x0$ is arbitrarily close
to $L$ (as close to $L$ as we like) for all $x$ sufficiently
close (but not equal) to $a$, we write

$$\
\lim_{x \to a} = L
$$

and say that the limit fo $f(x)$ as $x$ approaches $a$ equals $L$.
\enddef

\begdef{One-sided limits}
1. Right sided limits. Suppose $f$ is dsefined for all $x$
near $a$ with $x > a$. If $f(x)$ is sarbitrarily close $L$
for all $x$ sufficiently close to $a$ with $x > a$, we write

$$\
\lim_{x \to a^{+}} f(x) = L
$$

and say that the limit of $f(x)$ as $x$ approaches $a$
from the right equals $L$.

\par

2. Left-sided limit. Suppose $f$ is defined for all $x$
near $a$ with $x < a$. If $f(x)$ is arbitrarily close to $L$
for all x sufficiently clsoe to $a$ with $x < a$, we
write

$$\
\lim_{x \to a^{-}} = L
$$

and say the limit of $f(x)$ as $x$ approaches $a$ from
the left equals $L$.
\enddef

\begdef{Infinite Limits}
Suppose $f$ is defined for all $x$ near $a$. If $f(x)$
gorws arbitrarily large for all $x$ sufficiently close (but not equal)
to $a$, we write

$$\
\lim_{x \to a} f(x) = \infty
$$

and say the limit of $f(x)$ as $x$ approaches $a$ is
infinity.

\par If $f(x)$ is negative and grows arbitrarily large in magnitude for
all $x$ sufficiently close (but not equal to) $a$, we write

$$\
\lim_{x \to a} f(x) = -\infty
$$

and say the limit of $f(x)$ as $x$ approaches $a$ is
negative infinity.

{\it In both cases, the limit does not exist}.

\enddef

\begdef{Vertical Asymptote}
If
$\lim\limits_{x \to a} = \pm \infty$, 
$\lim\limits_{x \to a^+} = \pm \infty$, 
$\lim\limits_{x \to a^-} = \pm \infty$,
the line $x = a$ is called a {\bf vertical asymptote}
of $f$.
\enddef

\begdef{Limits at Infinity and Horizontal Asymptotes}
If $f(x)$ becomes arbitrarily close to a finite number $L$
for all sufficiently large and positive $x$, then we
write

$$\
\lim_{x \to \infty} = L
$$

We say the limit as $f(x)$ as $x$ approaches infinity
is $L$. in thise, the line $y = L$ is a {\bf horizontal asymptote}
of $f$. The limit at negative infinity, $\lim\limits_{x \to -\infty} f(x) = M$,
is defined analogously. When this limit exists, $y = M$ is a horizontal
asymptote.
\enddef

\begdef{Infinite Limits at Infinity}
If $f(x)$ becomes arbitrarily large as $x$ becomes arbitrarily
large, then we right

$$\
\lim_{x \to \infty} f(x) = \infty.
$$

The limits
$\lim\limits_{x \to \infty} = -\infty$,
$\lim\limits_{x \to -\infty} = \infty$,
and
$\lim\limits_{x \to -\infty} = -\infty$
are defined similarly.
\enddef

\begdef{Continuity at a Point}
A function $f$ is {\bf continuous} at $a$ if $\lim\limits_{x \to a}
f(x) = f(a)$.
\enddef

\begdef{Continuity at Endpoints}
A function $f$ is {\bf continuous from the right} (or
{\bf right-continuous} at $a$ if $\lim\limits_{x \to a^+} f(x) = f(a)$,
and $f$ is {\bf continuous from the left} (or {\bf left-continuous}) at $b$
if $\lim\limits_{x \to b^-} f(x) = f(b)$.
\enddef

\begdef{Continuity on an Interval}
A function $f$ is {\bf continuous on an interval $I$} if
it is continuous at all points of $I$. If $I$ contains its
endpoints, continuity on $I$ means continuous from the
right or left at the endpoints.
\enddef
