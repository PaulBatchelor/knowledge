\begdef{Rate of Change and Slope of the Tangent Line}
\enddef

The {\bf average rate of change} in $f$ on the interval
$[a, x]$ is the slope of the corresponding secant line:

$$\
m_{sec} = {f(x) - f(a) \over x - a}
$$

The {\bf instantaneous rate of change} in $f$ at $a$ is

$$\
m_{tan} = \lim_{x \to a} {
f(x) - f(a)
\over
x - a
}
$$

Which is also the {\bf slope of the tangent line} at $(a, f(a))$,
provided this limit exists. The {\bf tangent line} is the unique
line through $(a, f(a))$ with sloep $m_{tan}$. Its equation is:

$$\
y - f(a) = m_{tan}(x- a)
$$

\begdef {Alternate Definition: Rate of Change and the Slope of the Tangent Line}
The {\bf average rate of change} of $f$ on the interval $[a, a + h]$
is the slope of the corresponding secant line:

$$\
m_{sec} = {
f(a + h) - f(a)
\over
h
}
$$

The {\bf instantaneous rate of change} in $f$ at $a$ is

$$\
m_{tan} = \lim_{h \to 0}{
f(a + h) - f(a)
\over
h
}
$$

which is also the {\bf slope of the tangent line} at $(a, f(a))$,
provided this limit exists.

\enddef

\begdef{The derviative of a function at a point}
The {\bf derivative of $f$ at $a$}, denoted $f'(a)$, is given
by either of the two following limits, provided the limits exist and $a$ is in the
domain of $f$:

$$\
f'(a) = \lim_{x \to a}{
f(x) - f(a)
\over
x - a
}
$$

or

$$\
f'(a) = \lim_{h \to 0}{
f(a + h) - f(a)
\over
h
}
$$

\enddef

\begdef{The Derivative Function}
The {\bf derivative} is the function

$$\
f'(x) = \lim_{h \to 0}{
f(x + h) - f(x)
\over
h
}
$$

provided the limit exists and $x$ is in the domain of $f$.
If $f'(x)$ exists, we say that $f$ is {\bf differentiable}
at $x$. If $f$ is differentiable at every point of an open
interval $I$, we say that $f$ is differentiable from on $I$.

\enddef

\begdef{The Number $e$}
The number $e = 2.718281828459...$ satisfies

$$\
\lim_{h \to 0} {
e^h - 1
\over h
} = 1
$$

It is the base of the natural exponential function $f(x) = e^x$.
\enddef

\begdef{Higher-Order Derivatives}
Assuming $y = f(x)$ can be differentiated as often as necessary,
the {\bf second derivative} of $f$ is

$$\
f''(x) = {d \over dx}(f'(x)).
$$

For integers $n \ge 1$, the {\bf {\it n}th derivative} of $f$ is

$$\
f^{(n)}(x) = {d \over dx}(f^{(n - 1)}(x)).
$$

\enddef

\begdef{Limit of a Function (preliminary)}
Suppose the function $f$ is defined for all $x$ near $a$
except possibly at $a$. if $f(x0$ is arbitrarily close
to $L$ (as close to $L$ as we like) for all $x$ sufficiently
close (but not equal) to $a$, we write

$$\
\lim_{x \to a} = L
$$

and say that the limit fo $f(x)$ as $x$ approaches $a$ equals $L$.
\enddef

\begdef{One-sided limits}
1. Right sided limits. Suppose $f$ is dsefined for all $x$
near $a$ with $x > a$. If $f(x)$ is sarbitrarily close $L$
for all $x$ sufficiently close to $a$ with $x > a$, we write

$$\
\lim_{x \to a^{+}} f(x) = L
$$

and say that the limit of $f(x)$ as $x$ approaches $a$
from the right equals $L$.

\par

2. Left-sided limit. Suppose $f$ is defined for all $x$
near $a$ with $x < a$. If $f(x)$ is arbitrarily close to $L$
for all x sufficiently clsoe to $a$ with $x < a$, we
write

$$\
\lim_{x \to a^{-}} = L
$$

and say the limit of $f(x)$ as $x$ approaches $a$ from
the left equals $L$.
\enddef

\begdef{Infinite Limits}
Suppose $f$ is defined for all $x$ near $a$. If $f(x)$
gorws arbitrarily large for all $x$ sufficiently close (but not equal)
to $a$, we write

$$\
\lim_{x \to a} f(x) = \infty
$$

and say the limit of $f(x)$ as $x$ approaches $a$ is
infinity.

\par If $f(x)$ is negative and grows arbitrarily large in magnitude for
all $x$ sufficiently close (but not equal to) $a$, we write

$$\
\lim_{x \to a} f(x) = -\infty
$$

and say the limit of $f(x)$ as $x$ approaches $a$ is
negative infinity.

{\it In both cases, the limit does not exist}.

\enddef

\begdef{Vertical Asymptote}
If
$\lim\limits_{x \to a} = \pm \infty$, 
$\lim\limits_{x \to a^+} = \pm \infty$, 
$\lim\limits_{x \to a^-} = \pm \infty$,
the line $x = a$ is called a {\bf vertical asymptote}
of $f$.
\enddef

\begdef{Limits at Infinity and Horizontal Asymptotes}
If $f(x)$ becomes arbitrarily close to a finite number $L$
for all sufficiently large and positive $x$, then we
write

$$\
\lim_{x \to \infty} = L
$$

We say the limit as $f(x)$ as $x$ approaches infinity
is $L$. in thise, the line $y = L$ is a {\bf horizontal asymptote}
of $f$. The limit at negative infinity, $\lim\limits_{x \to -\infty} f(x) = M$,
is defined analogously. When this limit exists, $y = M$ is a horizontal
asymptote.
\enddef

\begdef{Infinite Limits at Infinity}
If $f(x)$ becomes arbitrarily large as $x$ becomes arbitrarily
large, then we right

$$\
\lim_{x \to \infty} f(x) = \infty.
$$

The limits
$\lim\limits_{x \to \infty} = -\infty$,
$\lim\limits_{x \to -\infty} = \infty$,
and
$\lim\limits_{x \to -\infty} = -\infty$
are defined similarly.
\enddef

\begdef{Continuity at a Point}
A function $f$ is {\bf continuous} at $a$ if $\lim\limits_{x \to a}
f(x) = f(a)$.
\enddef

\begdef{Continuity at Endpoints}
A function $f$ is {\bf continuous from the right} (or
{\bf right-continuous} at $a$ if $\lim\limits_{x \to a^+} f(x) = f(a)$,
and $f$ is {\bf continuous from the left} (or {\bf left-continuous}) at $b$
if $\lim\limits_{x \to b^-} f(x) = f(b)$.
\enddef

\begdef{Continuity on an Interval}
A function $f$ is {\bf continuous on an interval $I$} if
it is continuous at all points of $I$. If $I$ contains its
endpoints, continuity on $I$ means continuous from the
right or left at the endpoints.
\enddef

\begdef{Inverse properties for $e^x$ and $\ln x$}
{\par \bf 1.} $e^{\ln x} = x$, for $x > 0$, and $\ln(e^x) = x$, for all $x$.

{\par \bf 2.} $y = \ln x$ if and only if $x = e^x$.

{\par \bf 3.} For real numbers $x$ and $b > 0$, $b^x = e^{ln b^x} = e^{x \ln b}$

\enddef

\begdef{Absolute Maximum and Minimum}
Let $f$ be defined on a set $D$ containing $c$. If $f(c) \ge f(x)$
for every $x$ in $D$, then $f(c)$ is an {\bf absolute maximum value} of $f$
on $D$. If $f(c) \le f(x)$ for every $x$ in $D$, then
$f(c)$ is an {\bf absolute minimium} value of $f$ on $D$.
An ${\bf absolute extreme value}$ value is either an absolute
maximum value of an absolute minimum value.
\enddef

\begdef{Critical Point}
An interior point $c$ of the the domain of $f$ at which $f'(c) = 0$
fails to exist is called a {\bf critical point} of $f$.
\enddef

\begdef{Procedure for Locating Absolute Extreme Values on a Closed Interval}
Assume the function $f$ is continuous on the closed interval
$[a, b]$.

{\par \bf 1.}
Locate the critical points $c$ in $(a, b)$ where $f'(c) = 0$ or
$f'(c)$ does not exist. These points rae cnadidates for absolute
maxima and minima.

{\par \bf 2.}
Evaluate $f$ at the critical points and at the endpoints of $[a, b]$.

{\par \bf 3.}
Choose the larges and smallest values of $f$ from Step 2 for
the absolute maximum and minimum values, respectively.

\enddef

\begdef{Increasing and Decreasing Functions}
Suppose a function $f$ is defined on an interval $I$. we say that
$f$ is {\bf increasing} on $I$ if $f(x_2) > f(x_1)$ whenever
$x_1$ and $x_2$ are in $I$ and $x_2 > x_1$. We say that
$f$ is {\bf decreasing} on $I$ if $f(x_2) < f(x_1)$ whenever
$x_1$ and $x_2$ are in $I$ and $x_2 > x_1$.
\enddef

\begdef{Graphing Guidelines for $y = f(x)$}
{\par \bf 1. Identify the domain or interval of interest.}
On what interval(s) whould the funciton be graphed? It may
be the domain of the function or some subset of the domain.

{\par \bf 2. Exploit symmetry} 
Take advantage of symmetry. For example is the function even?
($f(-x) = f(x)$), odd ($f(-x) = -f(x)$, or neither?

{\par \bf 3. Find the first and second derivatives} 
They are needed to determine extrme avlues,
concavity, inflection points, and the intervals on whch $f$ is
increasing or decreasing. Computing derivatives, particularly
second derivatives, may not be practical, so some funciton may need
to be graphed without complete derivative information.

{\par \bf 4. Find critical points and possible inflection points}
Determine points wat which $f'(x) = 0$ or $f'$ is undefined.
Determine points at which $f''(x) = 0$ or $f''$ is undefined.

{\par \bf 5. Find intervals on which the function is increasing/decreasing
and concave up/down.}
The first derivate determines the intervals on which $f$ is
increasing or decreasing. The second derivative
determines the intervals on which the
function is concave up or concave down.

{\par \bf 6. Identify extreme values and inflection points.} 
Use either the first or second derivative test to classxify
the critcal points. Both x and y coordinates of maxima,
minima, and inflection points are needed for graphing.

{\par \bf 7. Locate all asymptotes and determine end behavior.} 
Vertical symptotse often occur at zeros of denominators.
Horizontal asyjmptotes require examining limits as $x \to \pm \infty$;
these limits determine end behavior. Slant asymptotes occur
with rational functions in which the degree of the numerator
is one more than the degree of the denominator.

{\par \bf 8. Find the intercepts.} 
The y-intercept of the graph is found by setting $x = 0$.
The x-intercepts are found by solving $f(x)$ = 0. They are the real zeros
(or roots) of $f$.

{\par \bf 9. Choose an appropriate graphing window and plot a graph.} 
Use the results opf the previous steps to graph the function.
If you use grpahing software, check for consistency with your
analytical work. Is your graph {\it complete}? That is,
does it show all th eessential details of the function?
\enddef

\begdef{Guidelines for Optimziation Problems}
{\par \bf 1.} Read the problem carefully, identify the
variables, and organize the given information with a picture.

{\par \bf 2.} Identify the objective function (the function to be
optimized). Write it in terms of the variables of the problem.

{\par \bf 3.} Identify the constraint(s). Write them in
terms of the variables of the problem.

{\par \bf 4.} Use the constraint(s) to eliminate all but one
independent variable of the objective function.

{\par \bf 5.} With the objective function expressed in terms
of a single variable, find the interval of interest for that
variable.

{\par \bf 6.} Use methods of calculus to find the absolute
maximum or minimum value of the objective function on
the interval of interest. If necessary, check the endpoints.

\enddef

\begdef{Linear Approximation of $f$ at $a$}
Suppose $f$ is differentiable on an interval $I$ containing
the point $a$. The {\bf linear approximation} to $f$ at
$a$ is the linear function

$$\
L(x) = f(a) + f'(a)(x -a), \, \hbox{for $x$ in $I$.}
$$
\enddef

\begdef{Relationship between $\Delta x$ and $\Delta y$}
Suppose $f$ is deifferntiable on an interval $I$ containing
the point $a$. The change in the value of $f$ between two points
$a$ and $a + \Delta x$ is approximately

$$\
\Delta y \approx f'(a) \Delta x
$$

where $a + \Delta x$ is in $I$.
\enddef

\begdef{Uses of Linear Approxmation}
To approximate $f$ near $x = a$, use

$$\
f(x) \approx L(x) = f(a) + f'(a)(x - a) \hbox{.}
$$

To approximate the change $\Delta y$ in the dependent variable
when the independent variable $x$ changes from $a$ to $a + \Delta x$,
use
$$\
\Delta y \approx f'(a)\Delta x
$$

\enddef

\begdef{Differentials}
Let $f$ be differentiable on an interval containing $x$.
A small change in $x$ is denoted by the {\bf differential} $dx$.
The corresponding change in $f$ is approximated by the
{\bf differential} $dy = f'(x) dx$; that is,

$$\
\Delta y = f(x + dx) - f(x) \approx dy = f'(x) dx
$$
\enddef

\begdef{Indeterminate forms $1^\infty$, $0^0$, and $\infty^0$}
Assume $\lim\limits_{x \to a}f(x)^{g(x)}$ has the indeterminate form
$1^{\infty}$, $0^0$, or $\infty^0$.

{\par \bf 1.}
Analyze $L = \lim\limits_{x \to a} g(x) \ln f(x)$.
This limit can be put into the form $0/0$ or $\infty / \infty$,
both of which are handled by l'Hopital's Rule.

{\par \bf 2.}
When $L$ is finite, $\lim\limits_{x \to a} f(x)^{g(x)} = e^L$.
If $L = \infty$ or $-\infty$, then
$\lim\limits_{x \to a} f(x)^{g(x)} = \infty$
or
$\lim\limits_{x \to a} f(x)^{g(x)} = 0$, respectively.
\enddef

\begdef{Growth Rates of Functions (as $x \to \infty$)}
Suppose $f$ and $g$ are functions with
$\lim\limits_{x \to \infty} f(x) =
\lim\limits_{x \to \infty} g(x) = \infty$.
Then
{\bf $f$ grows faster than $g$} as $x \to \infty$ if

$$\
\lim_{x \to \infty} {g(x) \over f(x)} = 0 \,
\hbox{or, equivalently, if}
\lim_{x \to \infty} {g(x) \over f(x)} = \infty
$$

The functions $f$ and $g$ have {\bf comparable growth rates}
if

$$\
\lim_{x \to infty} {f(x) \over g(x)} = M,
$$

where $0 < M < \infty$ ($M$ is positive and finite).
